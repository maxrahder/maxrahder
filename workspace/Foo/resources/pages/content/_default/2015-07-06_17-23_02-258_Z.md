Create your own criteria with +responsiveFormulas+.

<pre class="runnable">
Ext.create({
    xtype: 'viewport',
    items: [{
        xtype: 'button',
        plugins: ['responsive'],
        responsiveFormulas: {
            touch: function(context) {
                return Ext.feature.has.Touch;
            },
            notTouch: function(context) {
                return !Ext.feature.has.Touch;
            }
        },
        responsiveConfig: {
            touch: {
                text: 'Tap'
            },
            notTouch: {
                text: 'Click'
            }
        }
    }],
    renderTo: Ext.getBody()
});
</pre>
