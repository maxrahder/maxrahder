Ext.create('Ext.Container', {
    renderTo: Ext.getBody(),
    width: 500,
    height: 300,
    layout: 'fit',
    items: [{
        xtype: 'chart',
        store: getStore(),

        axes: [{
            type: 'numeric',
            position: 'left',
            title: 'Temperature (centigrade)',
            majorTickSteps: 4,
            minorTickSteps: 4,
            minimum: 0,
            maximum: 25
        }, {
            type: 'category',
            position: 'bottom',
            fields: ['month'],
            title: 'Month'
        }],
        series: [{
            type: 'line',
            axis: 'left',
            xField: 'month',
            yField: 'Paris'
        }, {
            type: 'line',
            axis: 'left',
            xField: 'month',
            yField: 'London'
        }]
    }]
});

function getStore() {
    Ext.define('MyApp.model.Sales', {
        extend: 'Ext.data.Model',
        fields: ['month', 'Paris', 'London']
    });
    var store = Ext.create('Ext.data.Store', {
        model: 'MyApp.model.Sales',
        proxy: {
            type: 'ajax',
            url: 'resources/charts/cityHighTemperatureByMonth.json'
        },
        autoLoad: true
    });

    return store;
}
