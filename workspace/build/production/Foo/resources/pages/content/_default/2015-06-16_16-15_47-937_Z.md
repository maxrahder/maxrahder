The +defaults+ config sets default values for child items.

<pre class="runnable">
Ext.define('MyApp.view.Main', {
    extend: 'Ext.container.Container',

    layout: 'vbox',
    defaults: {
        xtype: 'button',
        margin: 8
    },
    items: [{
        text: 'Button One'
    }, {
        text: 'Button Two'
    }]

});
Ext.application({
    name: 'MyApp',
    mainView: 'MyApp.view.Main'
});</pre>
