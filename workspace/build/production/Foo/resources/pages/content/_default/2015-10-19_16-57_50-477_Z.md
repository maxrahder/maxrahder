Ext.define('MyApp.view.Main', {
    extend: 'Ext.grid.Panel',

    columns : [ {
        xtype : 'rownumberer' 
    }, {
        text : 'Name',
        xtype : 'templatecolumn',
        tpl : '{first} {last}',
        dataIndex: 'last',
        flex : 1
    }],
    
    title : 'Aerosmith',
	store: {
        autoLoad: true,
	    fields : [ 'first', 'last', 'currentmember', 'networth' ],
        proxy : { type : 'ajax', url : 'resources/json/aerosmith.json' }
    },
});

Ext.application({
    name: 'MyApp',
    mainView: 'MyApp.view.Main'
});
