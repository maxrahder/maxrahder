On tablets, the input type determines the keyboard shown to the user. 
<pre class="runnable run 300">
Ext.define('MyApp.view.Main', {
    extend: 'Ext.form.Panel',
    items: [{
        fieldLabel: 'Email',
        inputType: 'email'
    }, {
        fieldLabel: 'Password',
        inputType: 'password'
    }, {
        fieldLabel: 'Repeat password',
        inputType: 'password'
    }],
    defaults: {
        width: 240,
        xtype: 'textfield'
    },
    title: 'inputType',
    bodyPadding: 8
});

Ext.application({
    name: 'MyApp',
    mainView: 'MyApp.view.Main'
});
