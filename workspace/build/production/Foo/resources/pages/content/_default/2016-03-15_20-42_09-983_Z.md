The items are any xtype you specify, decorated with the config for the
+Ext.tab.Tab+.

<pre class="runnable">
Ext.define('MyApp.view.Main', {
    extend: 'Ext.tab.Panel',
    items: [{
        xtype: 'component',
        html: 'Home', // A component config
        title: 'Home' // A tab button config
    }, {
        xtype: 'component',
        html: 'Users',
        title: 'Users'
    }]
});
Ext.application({
    name: 'MyApp',
    mainView: 'MyApp.view.Main'
});
</pre>
