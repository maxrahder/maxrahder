Fields with the same *name* are mutually exclusive. The selected value
is submitted with the form.

<pre class="runnable modern run">
Ext.define('MyApp.view.Main', {
    extend: 'Ext.form.Panel',
    items: [{
        xtype: 'radiofield',
        name: 'color',
        value: 'red',
        label: 'Red',
        checked: true
    }, {
        xtype: 'radiofield',
        name: 'color',
        value: 'green',
        label: 'Green'
    }, {
        xtype: 'radiofield',
        name: 'color',
        value: 'blue',
        label: 'Blue'
    }],
    defaults: {
        width: '30%',
        margin: 8
    },
    bodyPadding: 16
});
Ext.application({
    name: 'MyApp',
    mainView: 'MyApp.view.Main'
});

</pre>
