Buttons can have +Ext.menu.Item+ sub-items specified via the +menu:[]+. 
The property specifies the items in for an +Ext.menu.Menu+ container.

<pre class="runnable">Ext.define('MyApp.view.Main', {
    extend: 'Ext.panel.Panel',
    tbar: [{
        xtype: 'button',
        text: 'Beatles',
        
        menu: [{
            text: 'John',
            menu: [{
                text: 'Guitar'
            }, {
                text: 'Piano'
            }, {
                text: 'Harmonica'
            }, {
                text: 'Bass'
            }]
        }, {
            text: 'Paul',
            menu: [{
                text: 'Bass'
            }, {
                text: 'Guitar'
            }, {
                text: 'Piano'
            }, {
                text: 'Drums'
            }]
        }, {
            text: 'George',
            menu: [{
                text: 'Guitar'
            }, {
                text: 'Sitar'
            }]

        }, {
            text: 'Ringo',
            menu: [{
                text: 'Drums'
            }, {
                text: 'Tambourine'
            }, {
                text: 'Maracas'
            }]

        }]
    }],
});
Ext.application({
    name: 'MyApp',
    mainView: 'MyApp.view.Main'
});</pre>
