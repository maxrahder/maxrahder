    extend: 'Ext.Component',

    data: [{

        "name": "Bulbasaur"
    }],

    tpl: [
        '&lt;table>&lt;tr>',

        '&lt;tpl for=".">',

        '&lt;td>',
        '&lt;h1>{name}&lt;/h1>',

        '&lt;tpl if="img">',  // Start of if statement
        '&lt;img src="{img}"/>',
        '&lt;tpl else>',
        '&lt;img src="resources/json/pokemon/pokeball.png"/>',
        '&lt;/tpl>', // End else

        '&lt;/td>',

        '&lt;/tpl>',

        '&lt;/tr>&lt;/table>'
    ],
    padding: 8
});
Ext.application({
    name: 'MyApp',
    mainView: 'MyApp.view.Main'
});
